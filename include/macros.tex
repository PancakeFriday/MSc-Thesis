	}

\newcommand{\plt}[4][]{
	\begin{figure}[!htb]
		\ifthenelse{\equal{#1}{}}{}{\vspace{1em}}
		\begin{center}
			\input{plots/#2.pgf}
		\end{center}
		\ifthenelse{\equal{#1}{}}{}{
			\begin{textblock}{2}(0.2,-2.75)
				\textbf{a.}
			\end{textblock}
			\begin{textblock}{2}(5.7,-2.75)
				\textbf{b.}
			\end{textblock}
		}
		\setCaption{#3}{#4}
		\label{fig:#2}
	\end{figure}
}
